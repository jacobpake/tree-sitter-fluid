def zip3([x1, x2, x3], [y1, y2, y3]):

def zip3(x1 :| x2 :| x3 :| x4 :| xs, y1 :| y2 :| y3 :| y4 :| ys):
  (x1, y1) :| (x2, y2) :| (x3, y3) :| (x4, y4) :| zip3(xs, ys)
