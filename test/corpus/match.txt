def merge(xs, ys):
  match (xs, ys):
    case ([], _): ys
    case (x :| xs_, []): xs
    case (x :| xs_, y :| ys_):
      if x < y: x :| merge(xs_, ys)
      else: y :| merge(xs, ys_)
